SmartPark System

Colegio San Martín de Porres – La Serena, Coquimbo

¿En qué consiste nuestro proyecto?

Este sistema automatizado de estacionamiento tiene como objetivo mejorar la eficiencia, comodidad y control dentro de espacios vehiculares mediante tres circuitos distintos, cada uno diseñado para una función específica dentro del mismo sistema integrado.

Circuito 1: Control de acceso y conteo de espacios

Utiliza sensores infrarrojos para detectar entradas y salidas de vehículos, un servomotor para abrir/cerrar la barrera, y una pantalla LCD para mostrar los espacios disponibles. 

#include <Wire.h>
#include <LiquidCrystal_I2C.h>
#include <Servo.h>
LiquidCrystal_I2C lcd(0x27, 16, 2);
Servo servo;
const int IRentrada = 2;
const int IRsalida = 3;
int espacios = 5;
const int maxEspacios = 5;
int flag1 = 0;
int flag2 = 0;
void setup() {
  lcd.begin();
  lcd.backlight();
  servo.attach(4);
  pinMode(IRentrada, INPUT);
  pinMode(IRsalida, INPUT);
  lcd.setCursor(0, 0);
  lcd.print("Espacios:");
  lcd.setCursor(10, 0);
  lcd.print(espacios);
  servo.write(0);
}
void loop() {
  if (digitalRead(IRentrada) == LOW && flag1 == 0 && espacios > 0) {
    flag1 = 1;
    espacios--;
    servo.write(90);
    lcd.setCursor(10, 0);
    lcd.print(" ");
    lcd.setCursor(10, 0);
    lcd.print(espacios);
    delay(2000);
    servo.write(0);
  }
  if (digitalRead(IRentrada) == HIGH) flag1 = 0;
  if (digitalRead(IRsalida) == LOW && flag2 == 0 && espacios < maxEspacios) {
    flag2 = 1;
    espacios++;
    servo.write(90);
    lcd.setCursor(10, 0);
    lcd.print(" ");
    lcd.setCursor(10, 0);
    lcd.print(espacios);
    delay(2000);
    servo.write(0);
  }
  if (digitalRead(IRsalida) == HIGH) flag2 = 0;
}

Circuito 2: Luces RGB de ocupación

Este circuito indica con luces si un estacionamiento está disponible (verde) u ocupado (rojo) mediante un sensor IR individual por puesto. 

int sensorIR = 2;
int ledVerde = 3;
int ledRojo = 4;
void setup() {
  pinMode(sensorIR, INPUT);
  pinMode(ledVerde, OUTPUT);
  pinMode(ledRojo, OUTPUT);
}
void loop() {
  if (digitalRead(sensorIR) == LOW) {
    digitalWrite(ledRojo, HIGH);
    digitalWrite(ledVerde, LOW);
  } else {
    digitalWrite(ledRojo, LOW);
    digitalWrite(ledVerde, HIGH);
  }
}

Circuito 3: Mensaje de salida en LCD

Una pantalla LCD muestra un mensaje de despedida cada vez que un vehículo abandona el lugar, activado por un sensor IR. 

#include <Wire.h>
#include <LiquidCrystal_I2C.h>
LiquidCrystal_I2C lcd(0x27, 16, 2);
int sensorSalida = 2;
void setup() {
  lcd.begin();
  lcd.backlight();
  pinMode(sensorSalida, INPUT);
  lcd.setCursor(0, 0);
  lcd.print("SmartPark System");
}
void loop() {
  if (digitalRead(sensorSalida) == LOW) {
    lcd.clear();
    lcd.setCursor(0, 0);
    lcd.print("Gracias por su visita");
    delay(3000);
    lcd.clear();
    lcd.setCursor(0, 0);
    lcd.print("SmartPark System");
  }
}
